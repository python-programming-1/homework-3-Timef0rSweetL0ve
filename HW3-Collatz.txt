Write a Python function for the Collatz Conjecture.

def collatz(num):
    collatz_list = []
    while True:         # the while loop might generate and catch more than one result in each run
        try:
            if int(num) % 2 == 0 and int(num) > 1:  # even and positive integer
                num = int(num) // 2
                collatz_list.append(num)            # catch num into a list
            if int(num) % 2 == 1 and int(num) > 1:  # odd and positive integer
                num = 3 * int(num) + 1
                collatz_list.append(num)            # catch num into a list
            if int(num) > 1:                        # for positive integer only
                continue
            else:
                return collatz_list                 # retain the list value to be used in global scope
                break
        except ValueError:                          # for a non-integer value input
            print('Oops! Invalid. Input must be an integer.')
        break

print('Please type in an integer: ')
n = input()
result = collatz(n)
if result == []:                                    # for a negative integer has been input
    print('Only positive integer, plz!')
elif result != None:                                # for a positive integer
    print('Collatz conjecture from your input: ')
    for val in result:                              # retrieve each value from the list
        print(val)
