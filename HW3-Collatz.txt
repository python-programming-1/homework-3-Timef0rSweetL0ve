Write a Python function for the Collatz Conjecture.

def collatz(num):
    collatz_list = []
    while True:
        try:
            if int(num) % 2 == 0 and int(num) > 1:
                num = int(num) // 2
                collatz_list.append(num)   # catch num into a list
            if int(num) % 2 == 1 and int(num) > 1:
                num = 3 * int(num) + 1
                collatz_list.append(num)   # catch num into a list
            if int(num) > 1:
                continue
            else:
                return collatz_list
                break
        except ValueError:
            print('Oops! Invalid input... Please type in an integer: ')
        break

print('Please type in an integer: ')
n = input()
result = collatz(n)
if result == []:
    print('Please enter a positive number: ')
elif result != None:
    print('Collatz conjecture from your input: ')
    for val in result:   # retrieve each value from the list
        print(val)
